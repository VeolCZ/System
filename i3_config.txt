
# Use mod + \ to launch nautilus
bindsym $mod+backslash exec nautilus ~/Documents

# Brightness control
# NOTE: FIRST INSTALL brightnessctl
# NOTE 2: sudo usermod -a -G video veol
bindsym XF86MonBrightnessUp exec --no-startup-id brightnessctl s +5%
bindsym XF86MonBrightnessDown exec --no-startup-id brightnessctl s 5%-

# Launch applications on startup
exec --no-startup-id firefox
exec --no-startup-id code
exec --no-startup-id pavucontrol
exec --no-startup-id watch pkill dunst
exec --no-startup-id discord

# Assign applications to specific workspaces
assign [class="Firefox"] 1
assign [class="Code"] 2
assign [class="Discord"] 4
assign [class="Spotify"] 4
assign [class="Pavucontrol"] 10
